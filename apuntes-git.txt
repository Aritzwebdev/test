git clone: clonar repositorio de internet a git
git status: para ver si hay algo que añadir a git
git add: añadir algun archivo a git
git commit -am: aceptar cambio en git
git push origin master: subir fichero a internet
git pull: bajar fichero de intrernet

rm "nombre de archivo": borrar fichero
rmdir "nombre de directorio": borrar directorio

git config --global user.name "nombre de usuario"
git config --global user.email "email de usuario"
